033    package com.jeantessier.commandline;
034    
035    import java.util.*;
036    
037    public class TextPrinter extends Printer {
038        public TextPrinter(String command) {
039            append(command).eol();
040        }
041    
042        protected Set<String> getSwitchNames(CommandLine commandLine) {
043            return commandLine.getPresentSwitches();
044        }
045    
046        public void visitToggleSwitch(ToggleSwitch cls) {
047            indent().append("-").append(cls.getName()).eol();
048        }
049    
050        public void visitSingleValueSwitch(SingleValueSwitch cls) {
051            indent().append("-").append(cls.getName()).append(" ").append(cls.getValue()).eol();
052        }
053    
054        public void visitOptionalValueSwitch(OptionalValueSwitch cls) {
055            indent().append("-").append(cls.getName());
056            if (!"".equals(cls.getValue())) {
057                append(" ").append(cls.getValue());
058            }
059            eol();
060        }
061    
062        public void visitMultipleValuesSwitch(MultipleValuesSwitch cls) {
063            for (String value : (List<String>) cls.getValue()) {
064                indent().append("-").append(cls.getName()).append(" ").append(value).eol();
065            }
066        }
067    
068        public void visitAliasSwitch(AliasSwitch cls) {
069            // Do nothing
070        }
071    
072        public void visitNullParameterStrategy(NullParameterStrategy strategy) {
073            visitParameterStrategy(strategy);
074        }
075    
076        public void visitCollectingParameterStrategy(CollectingParameterStrategy strategy) {
077            visitParameterStrategy(strategy);
078        }
079    
080        public void visitAtLeastParameterStrategy(AtLeastParameterStrategy strategy) {
081            visitParameterStrategy(strategy);
082        }
083    
084        public void visitExactlyParameterStrategy(ExactlyParameterStrategy strategy) {
085            visitParameterStrategy(strategy);
086        }
087    
088        public void visitAtMostParameterStrategy(AtMostParameterStrategy strategy) {
089            visitParameterStrategy(strategy);
090        }
091    
092        private void visitParameterStrategy(ParameterStrategy strategy) {
093            for (String value : strategy.getParameters()) {
094                indent().append(value).eol();
095            }
096        }
097    }
