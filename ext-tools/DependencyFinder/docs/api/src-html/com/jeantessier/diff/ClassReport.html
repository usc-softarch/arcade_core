204        public void visitField_info(Field_info entry) {
205            if (entry.isPublic())     append(" visibility=\"public\"");
206            if (entry.isProtected())  append(" visibility=\"protected\"");
207            if (entry.isPackage())    append(" visibility=\"package\"");
208            if (entry.isPrivate())    append(" visibility=\"private\"");
209            if (entry.isStatic())     append(" static=\"yes\"");
210            if (entry.isFinal())      append(" final=\"yes\"");
211            if (entry.isVolatile())   append(" volatile=\"yes\"");
212            if (entry.isTransient())  append(" transient=\"yes\"");
213            if (entry.isSynthetic())  append(" synthetic=\"yes\"");
214            if (entry.isDeprecated()) append(" deprecated=\"yes\"");
215    
216            append(" type=\"").append(entry.getType()).append("\"");
217            append(" name=\"").append(entry.getName()).append("\"");
218            append(" signature=\"").append(entry.getSignature()).append("\"");
219            append(" full-signature=\"").append(entry.getFullSignature()).append("\"");
220    
221            if (entry.getConstantValue() != null) {
222                append(" value=\"");
223                entry.getConstantValue().accept(this);
224                append("\"");
225            }
226        }

396        public String render() {
397            raiseIndent();
398            raiseIndent();
399    
400            indent().append("<class>").eol();
401            raiseIndent();
402    
403            indent().append("<name>").append(differences.getName()).append("</name>").eol();
404    
405            if (differences.isDeclarationModified()) {
406                indent().append("<modified-declaration>").eol();
407                raiseIndent();
408    
409                indent().append("<old-declaration").append(breakdownDeclaration(differences.getOldClass())).append(">").append(differences.getOldDeclaration()).append("</old-declaration>").eol();
410                indent().append("<new-declaration").append(breakdownDeclaration(differences.getNewClass())).append(">").append(differences.getNewDeclaration()).append("</new-declaration>").eol();
411    
412                lowerIndent();
413                indent().append("</modified-declaration>").eol();
414            }
415    
416            if (removedFields.size() != 0) {
417                indent().append("<removed-fields>").eol();
418                raiseIndent();
419    
420                for (FeatureDifferences fd : removedFields) {
421                    indent();
422                    append("<declaration");
423                    fd.getOldFeature().accept(this);
424                    if (fd.isInherited()) {
425                        append(" inherited=\"yes\"");
426                    }
427                    append(">");
428                    append(fd.getOldDeclaration());
429                    append("</declaration>");
430                    eol();
431                }
432    
433                lowerIndent();
434                indent().append("</removed-fields>").eol();
435            }
436    
437            if (removedConstructors.size() != 0) {
438                indent().append("<removed-constructors>").eol();
439                raiseIndent();
440    
441                for (FeatureDifferences fd : removedConstructors) {
442                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getOldFeature())).append(fd.isInherited() ? " inherited=\"yes\"" : "").append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
443                }
444    
445                lowerIndent();
446                indent().append("</removed-constructors>").eol();
447            }
448    
449            if (removedMethods.size() != 0) {
450                indent().append("<removed-methods>").eol();
451                raiseIndent();
452    
453                for (FeatureDifferences fd : removedMethods) {
454                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getOldFeature())).append(fd.isInherited() ? " inherited=\"yes\"" : "").append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();

459            }

463                raiseIndent();

474                indent().append("<deprecated-constructors>").eol();
475                raiseIndent();
476    
477                for (FeatureDifferences fd : deprecatedConstructors) {
478                    indent().append("<declaration").append(breakdownDeclaration((Method_info) fd.getNewFeature())).append(">").append(fd.getOldDeclaration()).append("</declaration>").eol();
479                }

508                    raiseIndent();
509

512                    append("<old-declaration");

514                    append(">");
515                    if (fd.isConstantValueDifference()) {

518                        append(oldField.getDeclaration());
519                    }

522    
523                    Field_info newField = (Field_info) fd.getNewFeature();
524                    indent();
525                    append("<new-declaration");
526                    newField.accept(this);

532                    }

646                indent().append("<new-fields>").eol();
647                raiseIndent();
648    
649                for (FeatureDifferences fd : newFields) {
650                    indent();
651                    append("<declaration");
652                    fd.getNewFeature().accept(this);
653                    append(">");
654                    append(fd.getNewDeclaration());
655                    append("</declaration>");
656                    eol();
657                }
658    
659                lowerIndent();

661            }

663            if (newConstructors.size() != 0) {

667                for (FeatureDifferences fd : newConstructors) {

696        private String breakdownDeclaration(Classfile element) {
697            StringBuffer result = new StringBuffer();
698    
699            if (element != null) {
700                if (element.isPublic())     result.append(" visibility=\"public\"");
701                if (element.isPackage())    result.append(" visibility=\"package\"");
702                if (element.isFinal())      result.append(" final=\"yes\"");
703                if (element.isSuper())      result.append(" super=\"yes\"");
704                if (element.isSynthetic())  result.append(" synthetic=\"yes\"");
705                if (element.isDeprecated()) result.append(" deprecated=\"yes\"");
706    
707                result.append(" name=\"").append(element.getClassName()).append("\"");
708    
709                if (element.isInterface()) {
710                    result.append(" interface=\"yes\"");
711    
712                    result.append(" extends=\"");
713                    Iterator i = element.getAllInterfaces().iterator();
714                    while (i.hasNext()) {
715                        result.append(i.next());
716                        if (i.hasNext()) {
717                            result.append(", ");
718                        }
719                    }
720                    result.append("\"");
721                } else {
722                    if (element.isAbstract()) result.append(" abstract=\"yes\"");
723    
724                    result.append(" extends=\"").append(element.getSuperclassName()).append("\"");
725    
726                    result.append(" implements=\"");
727                    Iterator i = element.getAllInterfaces().iterator();

819        private String escapeXMLCharactersInTagContent(String text) {
820            String result = text;
821    
822            result = perl.substitute("s/&/&amp;/g", result);
823            result = perl.substitute("s/</&lt;/g", result);
824            result = perl.substitute("s/>/&gt;/g", result);
825    
826            return result;
827        }

829        private String escapeXMLCharactersInAttributeValue(String text) {
830            String result = escapeXMLCharactersInTagContent(text);
831    
832            result = perl.substitute("s/\"/&quot;/g", result);
833            result = perl.substitute("s/'/&apos;/g", result);
834    
835            return result;
836        }
