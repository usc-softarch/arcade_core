033    package com.jeantessier.dependency;
034    
035    import java.util.*;
036    
037    public class CollectionSelectionCriteria implements SelectionCriteria {
038        private boolean matchingPackages = true;
039        private boolean matchingClasses = true;
040        private boolean matchingFeatures = true;
041    
042        private Collection<String> include;
043        private Collection<String> exclude;
044    
045        public CollectionSelectionCriteria(Collection<String> include, Collection<String> exclude) {
046            this.include = include;
047            this.exclude = exclude;
048        }
049    
050        public boolean isMatchingPackages() {
051            return matchingPackages;
052        }
053    
054        public void setMatchingPackages(boolean matchingPackages) {
055            this.matchingPackages = matchingPackages;
056        }
057    
058        public boolean isMatchingClasses() {
059            return matchingClasses;
060        }
061    
062        public void setMatchingClasses(boolean matchingClasses) {
063            this.matchingClasses = matchingClasses;
064        }
065    
066        public boolean isMatchingFeatures() {
067            return matchingFeatures;
068        }
069    
070        public void setMatchingFeatures(boolean matchingFeatures) {
071            this.matchingFeatures = matchingFeatures;
072        }
073    
074        public boolean matches(PackageNode node) {
075            return matchesName(node.getName());
076        }
077    
078        public boolean matches(ClassNode node) {
079            return matchesName(node.getName());
080        }
081    
082        public boolean matches(FeatureNode node) {
083            return matchesName(node.getName());
084        }
085    
086        public boolean matchesPackageName(String name) {
087            return matchesName(name);
088        }
089    
090        public boolean matchesClassName(String name) {
091            return matchesName(name);
092        }
093    
094        public boolean matchesFeatureName(String name) {
095            return matchesName(name);
096        }
097    
098        private boolean matchesName(String name) {
099            return (include == null || include.contains(name)) && (exclude == null || !exclude.contains(name));
100        }
101    }
