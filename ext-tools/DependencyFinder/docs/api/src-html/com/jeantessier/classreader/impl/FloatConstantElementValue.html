033    package com.jeantessier.classreader.impl;
034    
035    import java.io.*;
036    
037    import org.apache.log4j.*;
038    
039    import com.jeantessier.classreader.*;
040    
041    public class FloatConstantElementValue extends ConstantElementValue implements com.jeantessier.classreader.FloatConstantElementValue {
042        public FloatConstantElementValue(ConstantPool constantPool, DataInput in) throws IOException {
043            super(constantPool, in);
044            Logger.getLogger(getClass()).debug("Const value index: " + getConstValueIndex() + " (" + getConstValue() + ")");
045        }
046    
047        public float getConstValue() {
048            return ((Float_info) getRawConstValue()).getValue();
049        }
050    
051        public char getTag() {
052            return ElementValueType.FLOAT.getTag();
053        }
054    
055        public void accept(Visitor visitor) {
056            visitor.visitFloatConstantElementValue(this);
057        }
058    }
