033    package com.jeantessier.classreader.impl;
034    
035    import java.io.*;
036    
037    import com.jeantessier.classreader.*;
038    
039    public class Integer_info extends ConstantPoolEntry implements com.jeantessier.classreader.Integer_info {
040        private int value;
041    
042        public Integer_info(ConstantPool constantPool, DataInput in) throws IOException {
043            super(constantPool);
044    
045            value = in.readInt();
046        }
047    
048        public int getValue() {
049            return value;
050        }
051    
052        public String toString() {
053            return String.valueOf(getValue());
054        }
055    
056        public int hashCode() {
057            return Integer.valueOf(getValue()).hashCode();
058        }
059    
060        public boolean equals(Object object) {
061            boolean result = false;
062    
063            if (this == object) {
064                result = true;
065            } else if (object != null && this.getClass().equals(object.getClass())) {
066                Integer_info other = (Integer_info) object;
067                result = this.getValue() == other.getValue();
068            }
069    
070            return result;
071        }
072    
073        public void accept(Visitor visitor) {
074            visitor.visitInteger_info(this);
075        }
076    }
