033    package com.jeantessier.dependencyfinder.cli;
034    
035    import java.io.*;
036    import java.util.*;
037    
038    import com.jeantessier.commandline.*;
039    import com.jeantessier.commandline.Printer;
040    import com.jeantessier.commandline.TextPrinter;
041    import com.jeantessier.dependencyfinder.*;
042    import com.jeantessier.dependency.*;
043    import org.apache.log4j.*;
044    
045    public abstract class Command {
046        public static final String DEFAULT_LOGFILE = "System.out";
047        public static final String DEFAULT_INCLUDES = "//";
048    
049        private CommandLine commandLine;
050        private CommandLineUsage commandLineUsage;
051    
052        private Date startTime;
053        private VerboseListener verboseListener;
054        private PrintWriter out;
055    
056        public String getName() {
057            return getClass().getSimpleName();
058        }
059    
060        private void resetCommandLine() {
061            commandLine = new CommandLine(getParameterStrategy());
062            populateCommandLineSwitches();
063        }
064    
065        protected ParameterStrategy getParameterStrategy() {
066            return new CollectingParameterStrategy();
067        }
068    
069        protected CommandLine getCommandLine() {
070            if (commandLine == null) {
071                resetCommandLine();
072            }
073    
074            return commandLine;
075        }
076    
077        public CommandLineUsage getCommandLineUsage() {
078            if (commandLineUsage == null) {
079                commandLineUsage = new CommandLineUsage(getName());
080                getCommandLine().accept(commandLineUsage);
081            }
082    
083            return commandLineUsage;
084        }
085    
086        protected VerboseListener getVerboseListener() {
087            return verboseListener;
088        }
089    
090        public void run(String[] args) throws Exception {
091            if (validateCommandLine(args, System.err)) {
092                process();
093            } else {
094                System.exit(1);
095            }
096        }
097    
098        protected void populateCommandLineSwitches() {
099            getCommandLine().addToggleSwitch("echo");
100            getCommandLine().addToggleSwitch("help");
101            getCommandLine().addSingleValueSwitch("out");
102            getCommandLine().addToggleSwitch("time");
103            getCommandLine().addOptionalValueSwitch("verbose", DEFAULT_LOGFILE);
104            getCommandLine().addToggleSwitch("version");
105        }
106    
107        protected void populateCommandLineSwitchesForXMLOutput(String defaultEncoding, String defaultDTDPrefix, String defaultIndentText) {
108            getCommandLine().addSingleValueSwitch("encoding", defaultEncoding);
109            getCommandLine().addSingleValueSwitch("dtd-prefix", defaultDTDPrefix);
110            getCommandLine().addSingleValueSwitch("indent-text", defaultIndentText);
111        }
112    
113        protected Collection<CommandLineException> parseCommandLine(String[] args) {
114            resetCommandLine();
115            return getCommandLine().parse(args);
116        }
117    
118        protected boolean validateCommandLine(String[] args, PrintStream out) {
119            boolean result = true;
120    
121            Collection<CommandLineException> exceptions = parseCommandLine(args);
122    
123            if (getCommandLine().getToggleSwitch("version")) {
124                showVersion(out);
125                result = false;
126            }
127    
128            if (getCommandLine().getToggleSwitch("help")) {
129                showError(out);
130                result = false;
131            }
132    
133            if (getCommandLine().getToggleSwitch("echo")) {
134                echo(out);
135                result = false;
136            }
137    
138            if (result) {
139                for (CommandLineException exception : exceptions) {
140                    result = false;
141                    Logger.getLogger(getClass()).error(exception);
142                }
143            }
144    
145            return result;
146        }
147    
148        protected Collection<CommandLineException> validateCommandLineForScoping() {
149            Collection<CommandLineException> exceptions = new ArrayList<CommandLineException>();
150    
151            if (hasScopeRegularExpressionSwitches() && hasScopeListSwitches()) {
152                exceptions.add(new CommandLineException("You can use switches for regular expressions or lists for scope, but not at the same time"));
153            }
154    
155            return exceptions;
156        }
157    
158        protected Collection<CommandLineException> validateCommandLineForFiltering() {
159            Collection<CommandLineException> exceptions = new ArrayList<CommandLineException>();
160    
161            if (hasFilterRegularExpressionSwitches() && hasFilterListSwitches()) {
162                exceptions.add(new CommandLineException("You can use switches for regular expressions or lists for filter, but not at the same time"));
163            }
164    
165            return exceptions;
166        }
167    
168        private void process() throws Exception {
169            startProcessing();
170            doProcessing();
171            stopProcessing();
172        }
173    
174        private void startProcessing() throws IOException {
175            startVerboseListener();
176            // Output is started lazily the first time it is requested.
177            startTimer();
178        }
179    
180        protected abstract void doProcessing() throws Exception;
181    
182        private void stopProcessing() throws IOException {
183            stopTimer();
184            stopOutput();
185            stopVerboseListener();
186        }
187    
188        private void startVerboseListener() throws IOException {
189            verboseListener = new VerboseListener();
190            if (commandLine.isPresent("verbose")) {
191                if (DEFAULT_LOGFILE.equals(commandLine.getOptionalSwitch("verbose"))) {
192                    verboseListener.setWriter(new OutputStreamWriter(System.out));
193                } else {
194                    verboseListener.setWriter(new FileWriter(commandLine.getOptionalSwitch("verbose")));
195                }
196            }
197        }
198    
199        private void stopVerboseListener() {
200            verboseListener.close();
201        }
202    
203        private void startTimer() {
204            startTime = new Date();
205        }
206    
207        private void stopTimer() {
208            if (commandLine.getToggleSwitch("time")) {
209                Date end = new Date();
210                System.err.println(getClass().getName() + ": " + ((end.getTime() - (double) startTime.getTime()) / 1000) + " secs.");
211            }
212        }
213    
214        private void startOutput() throws IOException {
215            if (getCommandLine().isPresent("out")) {
216                out = new PrintWriter(new FileWriter(getCommandLine().getSingleSwitch("out")));
217            } else {
218                out = new PrintWriter(new OutputStreamWriter(System.out));
219            }
220        }
221    
222        private void stopOutput() throws IOException {
223            if (out != null) {
224                out.close();
225            }
226        }
227    
228        protected void echo() {
229            echo(System.err);
230        }
231    
232        protected void echo(PrintStream out) {
233            Printer printer = new TextPrinter(getClass().getSimpleName());
234            getCommandLine().accept(printer);
235            out.println(printer);
236        }
237    
238        protected void showError() {
239            showError(System.err);
240        }
241    
242        protected void showError(PrintStream out) {
243            out.println(getCommandLineUsage());
244            showSpecificUsage(out);
245        }
246    
247        protected void showError(String msg) {
248            showError(System.err, msg);
249        }
250    
251        protected void showError(PrintStream out, String msg) {
252            out.println(msg);
253            showError(out);
254        }
255    
256        protected abstract void showSpecificUsage(PrintStream out);
257    
258        protected void showVersion() {
259            showVersion(System.err);
260        }
261    
262        protected void showVersion(PrintStream out) {
263            Version version = new Version();
264    
265            out.print(version.getImplementationTitle());
266            out.print(" ");
267            out.print(version.getImplementationVersion());
268            out.print(" (c) ");
269            out.print(version.getCopyrightDate());
270            out.print(" ");
271            out.print(version.getCopyrightHolder());
272            out.println();
273    
274            out.print(version.getImplementationURL());
275            out.println();
276    
277            out.print("Compiled on ");
278            out.print(version.getImplementationDate());
279            out.println();
280        }

282        protected void populateCommandLineSwitchesForScoping() {
283            populateRegularExpressionCommandLineSwitches("scope", true, DEFAULT_INCLUDES);
284            populateListCommandLineSwitches("scope");
285        }
286    
287        protected void populateCommandLineSwitchesForFiltering() {
288            populateRegularExpressionCommandLineSwitches("filter", true, DEFAULT_INCLUDES);
289            populateListCommandLineSwitches("filter");
290        }
291    
292        protected void populateCommandLineSwitchesForStartCondition() {
293            populateRegularExpressionCommandLineSwitches("start", false, DEFAULT_INCLUDES);
294            populateListCommandLineSwitches("start");
295        }
296    
297        protected void populateCommandLineSwitchesForStopCondition() {
298            populateRegularExpressionCommandLineSwitches("stop", false, null);
299            populateListCommandLineSwitches("stop");
300        }
301    
302        protected void populateRegularExpressionCommandLineSwitches(String name, boolean addToggles, String defaultIncludes) {
303            if (defaultIncludes != null) {
304                getCommandLine().addMultipleValuesSwitch(name + "-includes", defaultIncludes);
305            } else {
306                getCommandLine().addMultipleValuesSwitch(name + "-includes");
307            }
308            getCommandLine().addMultipleValuesSwitch(name + "-excludes");
309            getCommandLine().addMultipleValuesSwitch("package-" + name + "-includes");
310            getCommandLine().addMultipleValuesSwitch("package-" + name + "-excludes");
311            getCommandLine().addMultipleValuesSwitch("class-" + name + "-includes");
312            getCommandLine().addMultipleValuesSwitch("class-" + name + "-excludes");
313            getCommandLine().addMultipleValuesSwitch("feature-" + name + "-includes");
314            getCommandLine().addMultipleValuesSwitch("feature-" + name + "-excludes");
315    
316            if (addToggles) {
317                getCommandLine().addToggleSwitch("package-" + name);
318                getCommandLine().addToggleSwitch("class-" + name);
319                getCommandLine().addToggleSwitch("feature-" + name);
320            }
321        }
322    
323        protected void populateListCommandLineSwitches(String name) {
324            getCommandLine().addMultipleValuesSwitch(name + "-includes-list");
325            getCommandLine().addMultipleValuesSwitch(name + "-excludes-list");
326        }
327    
328        protected SelectionCriteria getScopeCriteria() {
329            return getSelectionCriteria("scope", new ComprehensiveSelectionCriteria());
330        }
331    
332        protected SelectionCriteria getFilterCriteria() {
333            return getSelectionCriteria("filter", new ComprehensiveSelectionCriteria());
334        }
335    
336        protected SelectionCriteria getStartCriteria() {
337            return getSelectionCriteria("start", new ComprehensiveSelectionCriteria());
338        }
339    
340        protected SelectionCriteria getStopCriteria() {
341            return getSelectionCriteria("stop", new NullSelectionCriteria());
342        }
343    
344        protected SelectionCriteria getSelectionCriteria(String name, SelectionCriteria defaultSelectionCriteria) {
345            SelectionCriteria result = defaultSelectionCriteria;
346    
347            if (hasRegularExpressionSwitches(name)) {
348                RegularExpressionSelectionCriteria regularExpressionFilterCriteria = new RegularExpressionSelectionCriteria();
349    
350                if (getCommandLine().isPresent("package-" + name) || getCommandLine().isPresent("class-" + name) || getCommandLine().isPresent("feature-" + name)) {
351                    regularExpressionFilterCriteria.setMatchingPackages(getCommandLine().getToggleSwitch("package-" + name));
352                    regularExpressionFilterCriteria.setMatchingClasses(getCommandLine().getToggleSwitch("class-" + name));
353                    regularExpressionFilterCriteria.setMatchingFeatures(getCommandLine().getToggleSwitch("feature-" + name));
354                }
355    
356                if (getCommandLine().isPresent(name + "-includes") || (!getCommandLine().isPresent("package-" + name + "-includes") && !getCommandLine().isPresent("class-" + name + "-includes") && !getCommandLine().isPresent("feature-" + name + "-includes"))) {
357                    // Only use the default if nothing else has been specified.
358                    regularExpressionFilterCriteria.setGlobalIncludes(getCommandLine().getMultipleSwitch(name + "-includes"));
359                }
360                regularExpressionFilterCriteria.setGlobalExcludes(getCommandLine().getMultipleSwitch(name + "-excludes"));
361                regularExpressionFilterCriteria.setPackageIncludes(getCommandLine().getMultipleSwitch("package-" + name + "-includes"));
362                regularExpressionFilterCriteria.setPackageExcludes(getCommandLine().getMultipleSwitch("package-" + name + "-excludes"));
363                regularExpressionFilterCriteria.setClassIncludes(getCommandLine().getMultipleSwitch("class-" + name + "-includes"));
364                regularExpressionFilterCriteria.setClassExcludes(getCommandLine().getMultipleSwitch("class-" + name + "-excludes"));
365                regularExpressionFilterCriteria.setFeatureIncludes(getCommandLine().getMultipleSwitch("feature-" + name + "-includes"));
366                regularExpressionFilterCriteria.setFeatureExcludes(getCommandLine().getMultipleSwitch("feature-" + name + "-excludes"));
367    
368                result = regularExpressionFilterCriteria;
369            } else if (hasListSwitches(name)) {
370                result = createCollectionSelectionCriteria(getCommandLine().getMultipleSwitch(name + "-includes-list"), getCommandLine().getMultipleSwitch(name + "-excludes-list"));
371            }
372            
373            return result;
374        }
375    
376        protected boolean hasScopeRegularExpressionSwitches() {
377            return hasRegularExpressionSwitches("scope");
378        }
379    
380        protected boolean hasFilterRegularExpressionSwitches() {
381            return hasRegularExpressionSwitches("filter");
382        }
383    
384        protected boolean hasRegularExpressionSwitches(String name) {
385            Collection<String> switches = getCommandLine().getPresentSwitches();
386    
387            return
388                switches.contains(name + "-includes") ||
389                switches.contains(name + "-excludes") ||
390                switches.contains("package-" + name) ||
391                switches.contains("package-" + name + "-includes") ||
392                switches.contains("package-" + name + "-excludes") ||
393                switches.contains("class-" + name) ||
394                switches.contains("class-" + name + "-includes") ||
395                switches.contains("class-" + name + "-excludes") ||
396                switches.contains("feature-" + name) ||
397                switches.contains("feature-" + name + "-includes") ||
398                switches.contains("feature-" + name + "-excludes");
399        }
400    
401        protected boolean hasScopeListSwitches() {
402            return hasListSwitches("scope");
403        }
404    
405        protected boolean hasFilterListSwitches() {
406            return hasListSwitches("filter");
407        }
408    
409        protected boolean hasListSwitches(String name) {
410            Collection<String> switches = getCommandLine().getPresentSwitches();
411    
412            return
413                switches.contains(name + "-includes-list") ||
414                switches.contains(name + "-excludes-list");
415        }
416    
417        protected CollectionSelectionCriteria createCollectionSelectionCriteria(Collection<String> includes, Collection<String> excludes) {
418            return new CollectionSelectionCriteria(loadCollection(includes), loadCollection(excludes));
419        }
420    
421        private Collection<String> loadCollection(Collection<String> filenames) {
422            Collection<String> result = null;
423    
424            if (!filenames.isEmpty()) {
425                result = new HashSet<String>();
426    
427                for (String filename : filenames) {
428                    BufferedReader reader = null;
429                    try {
430                        reader = new BufferedReader(new FileReader(filename));
431    
432                        String line;
433                        while ((line = reader.readLine()) != null) {
434                            result.add(line);
435                        }
436                    } catch (IOException ex) {
437                        Logger.getLogger(getClass()).error("Couldn't read file " + filename, ex);
438                    } finally {
439                        try {
440                            if (reader != null) {
441                                reader.close();
442                            }
443                        } catch (IOException ex) {
444                            Logger.getLogger(getClass()).error("Couldn't close file " + filename, ex);
445                        }
446                    }
447                }
448            }
449    
450            return result;
451        }
452    
453        protected PrintWriter getOut() throws IOException {
454            if (out == null) {
455                startOutput();
456            }
457    
458            return out;
459        }

462            this.out = out;
