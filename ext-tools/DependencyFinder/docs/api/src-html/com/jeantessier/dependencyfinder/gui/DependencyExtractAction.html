033    package com.jeantessier.dependencyfinder.gui;
034    
035    import java.awt.event.*;
036    import java.io.*;
037    import java.util.*;
038    
039    import javax.swing.*;
040    
041    import com.jeantessier.classreader.*;
042    import com.jeantessier.classreader.impl.*;
043    import com.jeantessier.dependency.*;
044    
045    public class DependencyExtractAction extends AbstractAction implements Runnable {
046        private DependencyFinder model;
047    
048        public DependencyExtractAction(DependencyFinder model) {
049            this.model = model;
050    
051            putValue(Action.LONG_DESCRIPTION, "Extract dependencies from compiled classes");
052            putValue(Action.NAME, "Extract");
053            putValue(Action.SMALL_ICON, new ImageIcon(getClass().getResource("icons/extract.gif")));
054        }
055    
056        public void actionPerformed(ActionEvent e) {
057            JFileChooser chooser;
058            if (model.getInputFiles().isEmpty()) {
059                chooser = new JFileChooser(new File("."));
060            } else {
061                chooser = new JFileChooser(model.getInputFiles().iterator().next());
062            }
063            chooser.addChoosableFileFilter(new JavaBytecodeFileFilter());
064            chooser.setFileSelectionMode(JFileChooser.FILES_AND_DIRECTORIES);
065            chooser.setMultiSelectionEnabled(true);
066            int returnValue = chooser.showDialog(model, "Extract");
067            if (returnValue == JFileChooser.APPROVE_OPTION) {
068                for (File file : chooser.getSelectedFiles()) {
069                    model.addInputFile(file);
070                }
071                new Thread(this).start();
072            }
073        }
074    
075        public void run() {
076            Date start = new Date();
077    
078            model.getStatusLine().showInfo("Scanning ...");
079            ClassfileScanner scanner = new ClassfileScanner();
080            scanner.load(model.getInputFiles());
081    
082            model.getProgressBar().setMaximum(scanner.getNbFiles());
083    
084            model.getMonitor().setClosedSession(false);
085    
086            ClassfileLoader loader = new TransientClassfileLoader(model.getClassfileLoaderDispatcher());
087            loader.addLoadListener(new VerboseListener(model.getStatusLine(), model.getProgressBar()));
088            loader.addLoadListener(model.getMonitor());
089            loader.load(model.getInputFiles());
090    
091            if (model.getMaximize()) {
092                model.getStatusLine().showInfo("Maximizing ...");
093                new LinkMaximizer().traverseNodes(model.getPackages());
094            } else if (model.getMinimize()) {
095                model.getStatusLine().showInfo("Minimizing ...");
096                new LinkMinimizer().traverseNodes(model.getPackages());
097            }
098    
099            Date stop = new Date();
100    
101            model.getStatusLine().showInfo("Done (" + ((stop.getTime() - start.getTime()) / (double) 1000) + " secs).");
102            model.setTitle("Dependency Finder - Extractor");
103        }
104    }
