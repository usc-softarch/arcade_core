033    package com.jeantessier.dependencyfinder.cli;
034    
035    import java.io.*;
036    
037    import com.jeantessier.classreader.*;
038    import com.jeantessier.dependency.*;
039    import com.jeantessier.dependencyfinder.*;
040    import com.jeantessier.metrics.*;
041    
042    public class VerboseListener extends VerboseListenerBase implements DependencyListener, MetricsListener {
043        private PrintWriter writer = new NullPrintWriter();
044    
045        public PrintWriter getWriter() {
046            return writer;
047        }
048    
049        public void setWriter(Writer writer) {
050            setWriter(new PrintWriter(writer));
051        }
052    
053        public void setWriter(PrintWriter writer) {
054            this.writer = writer;
055        }
056    
057        public void close() {
058            getWriter().close();
059        }
060    
061        public void print(String s) {
062            getWriter().println(s);
063        }
064    
065        public void beginSession(LoadEvent event) {
066            super.beginSession(event);
067    
068            getWriter().print("Searching for classes ...");
069            getWriter().println();
070            getWriter().flush();
071        }
072    
073        public void beginGroup(LoadEvent event) {
074            super.beginGroup(event);
075    
076            getWriter().print("Searching ");
077            getWriter().print(getCurrentGroup().getName());
078    
079            switch (getCurrentGroup().getSize()) {
080                case -1:
081                    break;
082    
083                case 0:
084                case 1:
085                    getWriter().print(" (");
086                    getWriter().print(getCurrentGroup().getSize());
087                    getWriter().print(" file)");
088                    break;
089    
090                default:
091                    getWriter().print(" (");
092                    getWriter().print(getCurrentGroup().getSize());
093                    getWriter().print(" files)");
094                    break;
095            }
096    
097            getWriter().print(" ...");
098            getWriter().println();
099            getWriter().flush();
100        }
101    
102        public void endClassfile(LoadEvent event) {
103            super.endClassfile(event);
104    
105            getWriter().print("Loading ");
106            getWriter().print(event.getClassfile());
107            getWriter().print(" ...");
108            getWriter().println();
109            getWriter().flush();
110        }
111    
112        public void endFile(LoadEvent event) {
113            super.endFile(event);
114    
115            if (!getVisitedFiles().contains(event.getFilename())) {
116                getWriter().print("Skipping ");
117                getWriter().print(event.getFilename());
118                getWriter().print(" ...");
119                getWriter().println();
120                getWriter().flush();
121            }
122        }
123    
124        public void beginSession(DependencyEvent event) {
125            // Do nothing
126        }
127    
128        public void beginClass(DependencyEvent event) {
129            getWriter().print("Getting dependencies from ");
130            getWriter().print(event.getClassName());
131            getWriter().print(" ...");
132            getWriter().println();
133            getWriter().flush();
134        }
135    
136        public void dependency(DependencyEvent event) {
137            // Do nothing
138        }
139    
140        public void endClass(DependencyEvent event) {
141            // Do nothing
142        }
143    
144        public void endSession(DependencyEvent event) {
145            // Do nothing
146        }
147    
148        public void beginSession(MetricsEvent event) {
149            // Do nothing
150        }
151    
152        public void beginClass(MetricsEvent event) {
153            getWriter().print("Computing metrics for ");
154            getWriter().print(event.getClassfile());
155            getWriter().print(" ...");
156            getWriter().println();
157            getWriter().flush();
158        }
159    
160        public void beginMethod(MetricsEvent event) {
161            // Do nothing
162        }
163    
164        public void endMethod(MetricsEvent event) {
165            // Do nothing
166        }
167    
168        public void endClass(MetricsEvent event) {
169            // Do nothing
170        }
171    
172        public void endSession(MetricsEvent event) {
173            // Do nothing
174        }
175    }
