033    package com.jeantessier.dependencyfinder.cli;
034    
035    import org.apache.log4j.*;
036    
037    import com.jeantessier.classreader.*;
038    import com.jeantessier.diff.*;
039    
040    public class ClassClassDiff extends DiffCommand {
041        public void doProcessing() throws Exception {
042            // Collecting data, first classfiles from JARs,
043            // then package/class trees using NodeFactory.
044    
045            ClassfileLoader oldJar = new AggregatingClassfileLoader();
046            oldJar.addLoadListener(getVerboseListener());
047            oldJar.load(getCommandLine().getMultipleSwitch("old"));
048    
049            ClassfileLoader newJar = new AggregatingClassfileLoader();
050            newJar.addLoadListener(getVerboseListener());
051            newJar.load(getCommandLine().getMultipleSwitch("new"));
052    
053            // Starting to compare, first at class level,
054            // then descending to feature level.
055    
056            Logger.getLogger(getClass()).info("Comparing ...");
057            getVerboseListener().print("Comparing ...");
058    
059            String name = getCommandLine().getSingleSwitch("name");
060            String oldLabel = getCommandLine().getSingleSwitch("old-label");
061            String newLabel = getCommandLine().getSingleSwitch("new-label");
062    
063            Classfile oldClass = oldJar.getAllClassfiles().iterator().next();
064            Classfile newClass = newJar.getAllClassfiles().iterator().next();
065    
066            Differences differences = getDifferencesFactory().createClassDifferences(name, oldClass, newClass);
067    
068            Logger.getLogger(getClass()).info("Printing results ...");
069            getVerboseListener().print("Printing results ...");
070    
071            Report report = new Report(getCommandLine().getSingleSwitch("encoding"), getCommandLine().getSingleSwitch("dtd-prefix"));
072            if (getCommandLine().isPresent("indent-text")) {
073                report.setIndentText(getCommandLine().getSingleSwitch("indent-text"));
074            }
075            report.setName(name);
076            report.setOldVersion(oldLabel);
077            report.setNewVersion(newLabel);
078    
079            differences.accept(report);
080            getOut().print(report.render());
081        }
082    
083        public static void main(String[] args) throws Exception {
084            new ClassClassDiff().run(args);
085        }
086    }
